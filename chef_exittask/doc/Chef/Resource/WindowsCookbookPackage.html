  super
  @provider = if Gem::Version.new(Chef::VERSION) >= Gem::Version.new('12.6.0')
                Chef::Provider::Package::Windows
              else
                Chef::Provider::WindowsCookbookPackage
              end

  Chef::Log.warn "Please use the package resource available in Chef Client 12.6+.\nwindows_package will be removed in the next major version release\nof the Windows cookbook.\n"
end
