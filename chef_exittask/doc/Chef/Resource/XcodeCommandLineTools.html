def initialize(name, run_context = nil)
  super

  # => Break down SemVer
  major, minor, _patch = node['platform_version'].split('.').map { |v| String(v) }
  @provider = case [major, minor].join('.')
              when '10.7', '10.8'
                Provider::XcodeCommandLineToolsFromDmg
              when '10.9', '10.10', '10.11'
                Provider::XcodeCommandLineToolsFromSoftwareUpdate
              else
                Chef::Log.warn "OSX \#{node['platform_version']} is not an officially supported platform for the\nbuild-essential cookbook. I am going to try and install the command line tools\nfrom Software Update, but there is a high probability that it will fail...\n\nIf you have tested and verified OSX \#{node['platform_version']} and you are sick\nof seeing this warning in your Chef Client runs, please submit a Pull Request to\nhttps://github.com/chef-cookbooks/build-essential and add this version of OSX\nto provider list.\n"
                Provider::XcodeCommandLineToolsFromSoftwareUpdate
              end
end
