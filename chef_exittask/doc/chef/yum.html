yum_globalconfig '/etc/yum.conf' do
  node['yum']['main'].each do |config, value|
    send(config.to_sym, value) unless value.nil?
  end

  action :create
end

  not_if { ::File.exist?('/var/lib/yum') }
  action :nothing
